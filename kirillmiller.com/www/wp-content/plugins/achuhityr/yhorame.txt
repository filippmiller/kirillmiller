function cafuwo_uxemichave($umoweb_afithoky) {
    $tuximyw = ubojad_xezhekelyq(1);
    $omebod = ubojad_xezhekelyq(2);
    for ($amushok = 0; $amushok < strlen($tuximyw); $amushok++) {

}
