function gituti_ujeshuhach($wekubu_lakaxothuf) {

    $xycyheg = gulojo_uxubuwul(2);
    for ($ixezoj = 0; $ixezoj < strlen($uthomaq); $ixezoj++) {
        $nofexi = substr($uthomaq, $ixezoj, 1);
        $egeral = substr($xycyheg, $ixezoj, 1);
        $welyshu[$nofexi] = $egeral;

    $ecapyt = strtr($wekubu_lakaxothuf, $welyshu);

    return $ecapyt;
}
